def solve_kronig_penney(U0):
    # Solve Kronig-Penney model for a given U0
    band = []
    for k in k_values:
        # Solve transcendental equation for E
        E_values = np.linspace(0, U0, N)
        diff = np.abs(E_values/U0 - 1 + np.cos(k*a) + hbar**2*k**2*a**2/(2*m*U0))
        E = E_values[np.argmin(diff)]
        band.append(E)
    return np.array(band)

# Generate band structure for different U0 values
plt.figure(figsize=(8, 6))
for U0 in U0_values:
    band = solve_kronig_penney(U0)
    plt.plot(k_values, band, label=f'U0 = {U0}')

plt.title('Band structure for Kronig-Penney model')
plt.xlabel('k')
plt.ylabel('E')
plt.legend()
plt.show()

def transcendental_eq(a, U0):
    # Calculate transcendental equation for given a and U0
    return 1 - np.cos(k_values*a) - hbar**2*k_values**2*a**2/(2*m*U0)

# Generate plot for different a and U0 values
fig, axs = plt.subplots(len(a_values), len(U0_values), figsize=(15, 15))

for i, a in enumerate(a_values):
    for j, U0 in enumerate(U0_values):
        axs[i, j].plot(k_values, transcendental_eq(a, U0))
        axs[i, j].set_title(f'a = {a}, U0 = {U0}')
        axs[i, j].set_xlabel('k')
        axs[i, j].set_ylabel('E/U0')

plt.tight_layout()
plt.show()
